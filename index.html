async function fetchData() { try { const response = await fetch("https://fkhrzzhd.github.io/REPOSITORY/data.json"); // Ganti dengan URL GitHub Pages Anda if (!response.ok) throw new Error("Gagal mengambil data"); const data = await response.json(); displayWords(data); } catch (error) { console.error("Error:", error); } } function displayWords(words) { const tableBody = document.getElementById("wordList"); tableBody.innerHTML = ""; if (!words || words.length === 0) { tableBody.innerHTML = "Data tidak ditemukan"; return; } words.forEach(word => { tableBody.innerHTML += ` ${word.kata} ${word.sinonim.join(", ")} ${word.kategori.join(", ")} ${word.asosiasi} `; }); } fetchData(); // Panggil saat halaman dimuat
